1use std::sync::Arc;
2
3use anyhow::Result;
4use async_trait::async_trait;
5use redis::aio::MultiplexedConnection;
6use redis::AsyncCommands;
7use tokio::sync::Mutex;
8
9use crate::set_trait::SeenSet;
10
11const DEFAULT_SEEN_KEY: &str = "argus:seen";
12
13#[derive(Clone)]
14pub struct RedisSeenSet {
15    conn: Arc<Mutex<MultiplexedConnection>>,
16    set_key: String,
17}
18
19impl RedisSeenSet {
20    pub fn new(conn: MultiplexedConnection, set_key: Option<String>) -> Self {
21        Self {
22            conn: Arc::new(Mutex::new(conn)),
23            set_key: set_key.unwrap_or_else(|| DEFAULT_SEEN_KEY.to_string()),
24        }
25    }
26
27    pub async fn connect(redis_url: &str, set_key: Option<String>) -> Result<Self> {
28        let client = redis::Client::open(redis_url)?;
29        let conn = client.get_multiplexed_tokio_connection().await?;
30        Ok(Self::new(conn, set_key))
31    }
32}
33
34#[async_trait]
35impl SeenSet for RedisSeenSet {
36    async fn insert_if_new(&self, url: String) -> bool {
37        let mut conn = self.conn.lock().await;
38        let key = self.set_key.clone();
39        let added: i32 = conn.sadd(&key, &url).await.unwrap_or(0);
40        added == 1
41    }
42}
