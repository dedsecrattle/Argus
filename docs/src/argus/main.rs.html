1use std::sync::Arc;
2
3use anyhow::{Context, Result};
4use argus_config::cli::{Cli, Command, CrawlOpts, SeedOpts};
5use argus_storage::{FileStorage, NoopStorage, Storage};
6use clap::Parser;
7use tracing_subscriber::{EnvFilter, FmtSubscriber};
8
9#[tokio::main]
10async fn main() -> Result<()> {
11    let cli = Cli::parse();
12
13    let subscriber = FmtSubscriber::builder()
14        .with_env_filter(
15            EnvFilter::try_from_default_env().unwrap_or_else(|_| EnvFilter::new("info")),
16        )
17        .finish();
18
19    tracing::subscriber::set_global_default(subscriber)?;
20
21    match cli.command {
22        Command::Crawl(opts) => run_crawl(opts).await,
23        Command::Seed(opts) => run_seed(opts).await,
24    }
25}

27async fn run_crawl(opts: CrawlOpts) -> Result<()> {
28    let seed_url = opts.seed_url;
29    let redis_url = opts.redis_url.as_deref();
30
31    if seed_url.is_none() && redis_url.is_none() {
32        anyhow::bail!("either --seed-url or --redis-url is required for crawl");
33    }
34    if seed_url.is_none() && redis_url.is_some() {
35        tracing::info!("running as consumer only (no seed URL)");
36    } else if let Some(ref u) = seed_url {
37        tracing::info!("starting argus with seed {}", u);
38    }
39
40    let config = argus_worker::worker::CrawlConfig {
41        seed_url,
42        max_depth: opts.max_depth,
43        global_concurrency: opts.global_concurrency,
44        per_host_concurrency: opts.per_host_concurrency,
45        per_host_delay_ms: opts.per_host_delay_ms,
46    };
47
48    let storage: Arc<dyn Storage> = match &opts.storage_dir {
49        Some(dir) => {
50            let s = FileStorage::new(dir);
51            s.ensure_dirs().await?;
52            Arc::new(s)
53        }
54        None => Arc::new(NoopStorage),
55    };
56
57    let shutdown = argus_worker::ShutdownSignal::new();
58    let shutdown_clone = shutdown.clone();
59
60    tokio::spawn(async move {
61        argus_worker::listen_for_shutdown(shutdown_clone).await;
62    });
63
64    if let Some(redis_url) = redis_url {
65        argus_worker::worker::run_redis(
66            config,
67            redis_url,
68            storage,
69            opts.redis_rate_limit,
70            Some(shutdown),
71        )
72        .await
73    } else {
74        let url = config
75            .seed_url
76            .as_deref()
77            .context("crawl without Redis requires --seed-url")?;
78        if url.is_empty() {
79            anyhow::bail!("--seed-url is required for in-memory crawl");
80        }
81        argus_worker::worker::run_in_memory(config, storage, Some(shutdown)).await
82    }
83}

85async fn run_seed(opts: SeedOpts) -> Result<()> {
86    let redis_url = opts
87        .redis_url
88        .as_deref()
89        .context("seed requires --redis-url")?;
90    if opts.url.is_empty() {
91        anyhow::bail!("seed requires at least one -u/--url");
92    }
93    argus_worker::worker::seed_redis(redis_url, &opts.url).await
94}
