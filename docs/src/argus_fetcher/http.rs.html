23            connect_timeout: Duration::from_secs(10),
24            request_timeout: Duration::from_secs(30),
25            max_redirects: 10,

31#[derive(Clone)]
32pub struct HttpFetcher {
33    client: Client,
34    config: FetcherConfig,
35}
36
37impl HttpFetcher {
38    pub fn new() -> Result<Self> {
39        Self::with_config(FetcherConfig::default())
40    }
41
42    pub fn with_config(config: FetcherConfig) -> Result<Self> {
43        let client = Client::builder()
44            .user_agent(&config.user_agent)
45            .redirect(reqwest::redirect::Policy::limited(config.max_redirects))
46            .connect_timeout(config.connect_timeout)
47            .timeout(config.request_timeout)
48            .build()?;
49
50        Ok(Self { client, config })
51    }
52
53    pub async fn fetch(&self, job: &CrawlJob) -> Result<FetchResult> {
54        let mut attempts = 0;
55        let max_retries = self.config.retry_config.max_retries;
56
57        loop {
58            match self.fetch_once(job).await {
59                Ok(result) => return Ok(result),
60                Err(err) => {
61                    if !err.is_retryable() || attempts >= max_retries {
62                        return Err(anyhow::anyhow!(err));
63                    }
64
65                    let backoff = self.config.retry_config.calculate_backoff(attempts);
66                    tracing::debug!(
67                        "retry attempt {} after {:?} for {}: {}",
68                        attempts + 1,
69                        backoff,
70                        job.url,
71                        err
72                    );
73
74                    tokio::time::sleep(backoff).await;
75                    attempts += 1;
76                }
77            }
78        }
79    }
80
81    async fn fetch_once(&self, job: &CrawlJob) -> Result<FetchResult, FetchError> {
82        let response = self
83            .client
84            .get(&job.url)
85            .send()
86            .await
87            .map_err(|e| FetchError::from_reqwest(&e))?;
88
89        let status = response.status().as_u16();
90        let final_url = response.url().to_string();
91        let content_type = response
92            .headers()
93            .get(reqwest::header::CONTENT_TYPE)
94            .and_then(|v| v.to_str().ok())
95            .map(|s| s.to_string());
96
97        let body = response
98            .bytes()
99            .await
100            .map_err(|e| FetchError::from_reqwest(&e))?;
101
102        Ok(FetchResult {
103            url: job.url.clone(),
104            final_url,
105            status,
106            content_type,
107            body,
108        })
109    }
110}
